create table workDay (
empno int,
workdate datetime
);

select max(workdate) from workday;

select now() from dual;

select  date_format(workdate,'%Y.%m.%d'),count(*) from workday where empno = 1
group by date_format(workdate,'%Y.%m.%d');


select  min(workdate) from workDay;
select  max(workdate) from workDay;
select  date_format(workdate,'%Y.%m.%d'), (select min(workdate) from workDay where date_format(e.workdate,'%Y.%m.%d') = date_format(workdate,'%Y.%m.%d')),
(select max(workdate) from workDay where date_format(e.workdate,'%Y.%m.%d') = date_format(workdate,'%Y.%m.%d'))
from workday e
group by e.date_format(workdate,'%Y.%m.%d');


select  date_format(workdate,'%Y.%m.%d'),  min(workdate) , max(workdate) 
from workday e
group by date_format(e.workdate,'%Y.%m.%d');

select  date_format(workdate,'%Y.%m.%d'),  min(workdate) , max(workdate) 
from workday e 
group by date_format(e.workdate,'%Y.%m.%d');

select  date_format(workdate,'%Y.%m.%d'), date_format( min(workdate), '%Y.%m.%d %T %p'), date_format( max(workdate), '%Y.%m.%d %T %p')
from workday e 
group by date_format(e.workdate,'%Y.%m.%d');

select  date_format(workdate,'%Y.%m.%d'), date_format( min(workdate), '%T %p'), date_format( max(workdate), '%T %p')
from workday e 
group by date_format(e.workdate,'%Y.%m.%d');
